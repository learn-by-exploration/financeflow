### Step 5: Implement `PUT /api/auth/password`

**File:** `src/routes/auth.js`

Add before `return router;`:

```js
// PUT /api/auth/password
router.put('/password', requireAuth, (req, res, next) => {
  try {
    const { current_password, new_password } = req.body;
    if (!current_password || !new_password) {
      return res.status(400).json({
        error: { code: 'VALIDATION_ERROR', message: 'current_password and new_password required' }
      });
    }
    if (new_password.length < 8) {
      return res.status(400).json({
        error: { code: 'VALIDATION_ERROR', message: 'New password must be at least 8 characters' }
      });
    }

    const user = db.prepare('SELECT * FROM users WHERE id = ?').get(req.user.id);
    if (!bcrypt.compareSync(current_password, user.password_hash)) {
      return res.status(401).json({
        error: { code: 'UNAUTHORIZED', message: 'Current password is incorrect' }
      });
    }

    const hash = bcrypt.hashSync(new_password, config.auth.saltRounds);

    const rotateTx = db.transaction(() => {
      db.prepare('UPDATE users SET password_hash = ?, updated_at = datetime(\'now\') WHERE id = ?')
        .run(hash, req.user.id);
      // Invalidate ALL sessions (session rotation)
      db.prepare('DELETE FROM sessions WHERE user_id = ?').run(req.user.id);
      // Issue new session
      const token = crypto.randomBytes(32).toString('hex');
      const expiresAt = new Date(Date.now() + config.session.maxAgeDays * 86400000).toISOString();
      db.prepare('INSERT INTO sessions (user_id, token, expires_at) VALUES (?, ?, ?)')
        .run(req.user.id, token, expiresAt);
      return token;
    });

    const newToken = rotateTx();
    audit.log(req.user.id, 'user.password_change', 'user', req.user.id);
    res.json({ token: newToken, message: 'Password changed successfully' });
  } catch (err) { next(err); }
});
```

Key design decisions:
- **Transaction wrapping:** password update + session invalidation + new session creation are atomic
- **Session rotation:** ALL sessions deleted (including current), new token returned — caller MUST use new token
- **bcrypt cost factor:** uses existing `config.auth.saltRounds` (12)
- **Audit logging:** records `user.password_change` event

### Step 6: Implement `DELETE /api/auth/account`

**File:** `src/routes/auth.js`

Add before `return router;`:

```js
// DELETE /api/auth/account
router.delete('/account', requireAuth, (req, res, next) => {
  try {
    const { password } = req.body;
    if (!password) {
      return res.status(400).json({
        error: { code: 'VALIDATION_ERROR', message: 'Password required for account deletion' }
      });
    }

    const user = db.prepare('SELECT * FROM users WHERE id = ?').get(req.user.id);
    if (!bcrypt.compareSync(password, user.password_hash)) {
      return res.status(401).json({
        error: { code: 'UNAUTHORIZED', message: 'Incorrect password' }
      });
    }

    // CASCADE DELETE handles all child data (sessions, accounts, transactions,
    // categories, budgets, goals, subscriptions, settings, tags, recurring_rules,
    // financial_health_scores, net_worth_snapshots, audit_log (SET NULL))
    db.prepare('DELETE FROM users WHERE id = ?').run(req.user.id);

    res.json({ ok: true, message: 'Account deleted' });
  } catch (err) { next(err); }
});
```

Key design decisions:
- **No transaction needed:** single DELETE statement; SQLite foreign key cascades handle all child rows
- **Relies on schema `ON DELETE CASCADE`:** verified that all user-owned tables (sessions, accounts, transactions, categories, budgets, savings_goals, subscriptions, groups, recurring_rules, tags, settings, net_worth_snapshots, financial_health_scores) cascade from `users(id)`
- **Audit log:** uses `ON DELETE SET NULL` — preserves audit entries with `user_id = NULL` for forensic traceability
- **No soft-delete:** permanent, irreversible. Password confirmation is the gate.

## 6. Risk Assessment

| Risk | Likelihood | Mitigation |
|------|-----------|------------|
| `ON DELETE CASCADE` not firing | Low (pragma `foreign_keys = ON` is set in db/index.js) | Cascade test explicitly verifies all child tables |
| bcrypt timing allows enumeration | Low (compareSync is constant-time for valid hashes) | Rate limiter reduces attempt volume |
| Test isolation — shared test user deleted | Medium | Tests use `beforeEach(() => cleanDb())` and create fresh users per test; never delete the shared testuser |
| Rate limiter interferes with other tests | Low | Auth limiter is skipped in test mode (`config.isTest` check) |
